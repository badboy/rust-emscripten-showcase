Embed into HTML

Code:

in-html.rs:

fn main() {
    println!("Hello Rust!");
}

site.html:

<!DOCTYPE html>
<html lang="de">
  <head>
    <meta charset="utf-8" />
    <title>Loading Rust code</title>
    <link rel="stylesheet" href="" />
  </head>
  <body>
    <script src="in-html.js"></script>
  </body>
</html>

Compile

rustc --target asmjs-unknown-emscripten in-html.rs

Run

Check the console of this site.
